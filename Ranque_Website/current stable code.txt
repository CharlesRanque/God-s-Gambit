      ultRefreshed: false,
      timerInterval: null,
      enemyTimerInterval: null
    };
    for (let i = 0; i < 5; i++) drawCard();
    const sampleEnemy = opposingFaction === 'heroes'
      ? { ...BRONZE_UNITS_HEROES[0], id: 'en1_init', hasAttacked: false, playOrder: enemyPlayOrderCounter++, passive: 'Guard' }
      : { ...BRONZE_UNITS_MONSTERS[3], id: 'en1_init', hasAttacked: false, playOrder: enemyPlayOrderCounter++, passive: 'Thorns 1' };
    if (!sampleEnemy.passive) assignPassive(sampleEnemy);
    state.enemyMelee[2] = sampleEnemy;

    document.getElementById('menuScreen').style.display = 'none';
    document.getElementById('mainMenu').style.display = 'none';
    document.getElementById('gameInterface').classList.add('active');
    document.getElementById('leftPanelOuter').style.display = 'flex';
    document.getElementById('rightPanelOuter').style.display = 'flex';
    document.getElementById('diceInfo').textContent = 'Every 2 Rounds';
    document.getElementById('enemyTimer').textContent = '30s';

    renderAll();
    startTimer();
    triggerAutoOverlay('Match Started');
    setupPointerEngine();
    checkAffordableCards();
  }

  function drawCard() {
    if (!state.deck.length) return;
    let card = state.deck.pop();
    card._isNewDraw = true;
    if (!card.passive && card.type !== 'Spell') assignPassive(card);
    state.hand.push(card);
  }

  function startTimer() {
    if (state.timerInterval) clearInterval(state.timerInterval);
    state.turnTimer = 30;
    const timerElement = document.getElementById('turnTimer');
    timerElement.classList.remove('critical-danger');
    timerElement.textContent = `${state.turnTimer}s`;
    state.timerInterval = setInterval(() => {
      state.turnTimer--;
      timerElement.textContent = `${state.turnTimer}s`;
      if (state.turnTimer <= 10) timerElement.classList.add('critical-danger');
      else timerElement.classList.remove('critical-danger');
      if (state.turnTimer <= 0 && state.turn === 'player' && !state.gameOver) endTurn();
    }, 1000);
  }

  function hasEnemyUnits() { return state.enemyMelee.some(u => u !== null) || state.enemyRanged.some(u => u !== null); }

  function checkAffordableCards() {
    if (state.turn !== 'player' || state.gameOver) return;
    const availableProv = state.provPool - state.provUsed;
    const hasAffordable = state.hand.some(c => c.cost <= availableProv);
    renderAll();
    if (!hasAffordable && state.hand.length > 0) {
      document.getElementById('logMessage').textContent = '⚠️ No affordable cards! Auto-ending turn...';
      setTimeout(() => { if (state.turn === 'player' && !state.gameOver) endTurn(); }, 1500);
    }
  }

    }, 1500);
  }

  function applyDiceEffect(roll) {
    switch(roll) {
      case 1: setTimeout(() => {
        let playerUnits = [];
        state.playerMelee.forEach((c, i) => { if(c) playerUnits.push({card: c, arr: state.playerMelee, idx: i}); });
        state.playerRanged.forEach((c, i) => { if(c) playerUnits.push({card: c, arr: state.playerRanged, idx: i}); });
        if (playerUnits.length > 0) {
          playerUnits.sort((a, b) => (a.card.playOrder || 999) - (b.card.playOrder || 999));
          const oldest = playerUnits[0];
          state.playerGrave.push({...oldest.card, hp: oldest.card.maxHp});
          oldest.arr[oldest.idx] = null;
        }
        let enemyUnits = [];
        state.enemyMelee.forEach((c, i) => { if(c) enemyUnits.push({card: c, arr: state.enemyMelee, idx: i}); });
        state.enemyRanged.forEach((c, i) => { if(c) enemyUnits.push({card: c, arr: state.enemyRanged, idx: i}); });
        if (enemyUnits.length > 0) {
          enemyUnits.sort((a, b) => (a.card.playOrder || 999) - (b.card.playOrder || 999));
          const oldest = enemyUnits[0];
          state.enemyGrave.push({...oldest.card, hp: oldest.card.maxHp});
          oldest.arr[oldest.idx] = null;
        }
        document.getElementById('logMessage').textContent = "💀 Reaper's Toll: Oldest units destroyed!";
        renderAll();
      }, 1000); break;
      case 2: setTimeout(() => {
        let pAll = [];
        state.playerMelee.forEach((c, i) => { if(c) pAll.push({card: c, arr: state.playerMelee, idx: i, row: 'melee'}); });
        state.playerRanged.forEach((c, i) => { if(c) pAll.push({card: c, arr: state.playerRanged, idx: i, row: 'ranged'}); });
        if (pAll.length > 0) {
          const pRand = pAll[Math.floor(Math.random() * pAll.length)];
          const targetArr = pRand.row === 'melee' ? state.playerRanged : state.playerMelee;
          const emptySlot = targetArr.findIndex(s => s === null);
          if (emptySlot !== -1) { targetArr[emptySlot] = pRand.card; pRand.arr[pRand.idx] = null; }
        }
        let eAll = [];
        state.enemyMelee.forEach((c, i) => { if(c) eAll.push({card: c, arr: state.enemyMelee, idx: i, row: 'melee'}); });
        state.enemyRanged.forEach((c, i) => { if(c) eAll.push({card: c, arr: state.enemyRanged, idx: i, row: 'ranged'}); });
        if (eAll.length > 0) {
          const eRand = eAll[Math.floor(Math.random() * eAll.length)];
          const targetArr = eRand.row === 'melee' ? state.enemyRanged : state.enemyMelee;
          const emptySlot = targetArr.findIndex(s => s === null);
          if (emptySlot !== -1) { targetArr[emptySlot] = eRand.card; eRand.arr[eRand.idx] = null; }
        }
        document.getElementById('logMessage').textContent = '🔄 Shifting Sands: Units switched rows!';
        renderAll();
      }, 1000); break;
      case 3: (function() {
        let eRanged = []; let eMelee = [];
        state.enemyRanged.forEach((c, i) => { if(c) eRanged.push({card: c, idx: i}); });
        state.enemyMelee.forEach((c, i) => { if(c) eMelee.push({card: c, idx: i}); });
        if (eRanged.length > 0 && eMelee.length > 0) {
          const r = eRanged[Math.floor(Math.random() * eRanged.length)];
          const m = eMelee[Math.floor(Math.random() * eMelee.length)];
          if (r.card.atk >= m.card.hp) {
            state.enemyGrave.push({...m.card, hp: m.card.maxHp});
            state.enemyMelee[m.idx] = null;
          } else if (m.card.atk >= r.card.hp) {
            state.enemyGrave.push({...r.card, hp: r.card.maxHp});
            state.enemyRanged[r.idx] = null;
          }
          document.getElementById('logMessage').textContent = '💥 Crossfire: Enemy units fought each other!';
          renderAll();
        }
      })(); break;
      case 4: setTimeout(() => {
        state.godHp = Math.max(0, state.godHp - 2);
        state.enemyGodHp = Math.max(0, state.enemyGodHp - 2);
        executeShake();
        document.getElementById('logMessage').textContent = '🩸 Blood Price: Both Gods lost 2 HP!';
        renderAll();
      }, 800); break;
      case 5: setTimeout(() => {
        if (state.playerGrave.length >= 2 && state.enemyGrave.length >= 2) {
          const pCard = state.playerGrave.splice(Math.floor(Math.random() * state.playerGrave.length), 1)[0];
          state.hand.push({...pCard, _isNewDraw: true});
          const eCard = state.enemyGrave.splice(Math.floor(Math.random() * state.enemyGrave.length), 1)[0];
          state.enemyDeckSize++;
          document.getElementById('logMessage').textContent = '👻 Lingering Soul: Each player returned 1 card from graveyard!';
        } else {
          document.getElementById('logMessage').textContent = '👻 Lingering Soul: Not enough cards in graveyards (need at least 2 each).';
        }
        renderAll();
      }, 1000); break;
      case 6: setTimeout(() => {
        let pUnits = [];
        state.playerMelee.forEach((c, i) => { if(c) pUnits.push({card: c, arr: state.playerMelee, idx: i}); });
        state.playerRanged.forEach((c, i) => { if(c) pUnits.push({card: c, arr: state.playerRanged, idx: i}); });
        if (pUnits.length > 0) {
          const pTarget = pUnits[Math.floor(Math.random() * pUnits.length)];
          pTarget.card.hp -= 2;
          if (pTarget.card.hp <= 0) { state.playerGrave.push({...pTarget.card, hp: pTarget.card.maxHp}); pTarget.arr[pTarget.idx] = null; }
        }
        let enUnits = [];
        state.enemyMelee.forEach((c, i) => { if(c) enUnits.push({card: c, arr: state.enemyMelee, idx: i}); });
        state.enemyRanged.forEach((c, i) => { if(c) enUnits.push({card: c, arr: state.enemyRanged, idx: i}); });
        if (enUnits.length > 0) {
          const eTarget = enUnits[Math.floor(Math.random() * enUnits.length)];
          eTarget.card.hp -= 2;
          if (eTarget.card.hp <= 0) { state.enemyGrave.push({...eTarget.card, hp: eTarget.card.maxHp}); eTarget.arr[eTarget.idx] = null; }
        }
        document.getElementById('logMessage').textContent = '⚡ Static Shock: Random units took 2 damage!';
        renderAll();
      }, 1000); break;
    }
  }

  function setupPointerEngine() {
    const svgLayer = document.getElementById('arrowSvgLayer');
    const arrowLine = document.getElementById('arrowLine');
    const attackIndicator = document.getElementById('attackIndicator');

    document.addEventListener('mousedown', (e) => {
      spawnClickRipple(e.clientX, e.clientY);
      if (state.gameOver || state.turn !== 'player') return;

      const boardCard = e.target.closest('.player-slot .card');
      if (boardCard && !boardCard.classList.contains('has-attacked')) {
        const slot = boardCard.closest('.slot');
        if (!slot || !slot.dataset.row || slot.dataset.idx === undefined) return;
        selectedAttacker = { row: slot.dataset.row, idx: parseInt(slot.dataset.idx) };
        document.querySelectorAll('.card.selected-for-attack').forEach(c => c.classList.remove('selected-for-attack'));
        boardCard.classList.add('selected-for-attack');
        document.getElementById('logMessage').textContent = '🎯 Card selected! Click an enemy to attack.';
        spawnActionSpark(e.clientX, e.clientY, '🎯');
        dragTracker = { active: true, type: 'combat', origin: { row: slot.dataset.row, idx: parseInt(slot.dataset.idx) }, startX: e.clientX, startY: e.clientY };
        svgLayer.style.display = 'block';
        updateArrow(e.clientX, e.clientY);
        return;
      }

      if (selectedAttacker && !selectedAttacker.isHandCard && (e.target.closest('.enemy-slot') || e.target.closest('#enemyGodHpDisplay') || e.target.closest('#enemyGodSide'))) {
       const target = e.target.closest('.enemy-slot') || e.target.closest('#enemyGodHpDisplay') || e.target.closest('#enemyGodSide');
        if (target) { resolveDragAction(target); selectedAttacker = null; document.querySelectorAll('.card.selected-for-attack').forEach(c => c.classList.remove('selected-for-attack')); svgLayer.style.display = 'none'; attackIndicator.style.display = 'none'; return; }
      }
      if (selectedAttacker && !selectedAttacker.isHandCard) { selectedAttacker = null; document.querySelectorAll('.card.selected-for-attack').forEach(c => c.classList.remove('selected-for-attack')); document.getElementById('logMessage').textContent = 'Click a card to attack, or drag to play!'; }

      const handCard = e.target.closest('.fan-card-wrapper');
      if (handCard) {
        const idx = parseInt(handCard.dataset.idx);
        const cardData = state.hand[idx];
        if (cardData && !handCard.querySelector('.card.unaffordable')) {
          if (cardData.type === 'Unit') {
            document.querySelectorAll('.card.selected-for-attack').forEach(c => c.classList.remove('selected-for-attack'));
            handCard.querySelector('.card').classList.add('selected-for-attack');
            selectedAttacker = { handIdx: idx, card: cardData, isHandCard: true };
            document.getElementById('logMessage').textContent = '🎯 Card selected! Click an empty player slot to play.';
            spawnActionSpark(e.clientX, e.clientY, '🎯');
            return;
          }
          dragTracker = { active: true, type: 'spell', origin: { handIdx: idx, card: cardData }, startX: e.clientX, startY: e.clientY };
          svgLayer.style.display = 'block';
          updateArrow(e.clientX, e.clientY);
          return;
        }
      }
      if (selectedAttacker && selectedAttacker.isHandCard && e.target.closest('.player-slot')) {
        const targetSlot = e.target.closest('.player-slot');
        if (targetSlot) { playCardFromHandToSlot(targetSlot, selectedAttacker); selectedAttacker = null; document.querySelectorAll('.card.selected-for-attack').forEach(c => c.classList.remove('selected-for-attack')); return; }
      }

      const handCardDrag = e.target.closest('.fan-card-wrapper');
      if (handCardDrag && !selectedAttacker?.isHandCard) {
        const idx = parseInt(handCardDrag.dataset.idx);
        const cardData = state.hand[idx];
        if (cardData && !handCardDrag.querySelector('.card.unaffordable')) {
          dragTracker = { active: true, type: cardData.type === 'Spell' ? 'spell' : 'play-unit', origin: { handIdx: idx, card: cardData }, startX: e.clientX, startY: e.clientY };
          svgLayer.style.display = 'block'; updateArrow(e.clientX, e.clientY);
        }
      }
    });

    document.addEventListener('mousemove', (e) => {
      if (!dragTracker.active) { attackIndicator.style.display = 'none'; return; }
      updateArrow(e.clientX, e.clientY);
      if (dragTracker.type === 'combat') {
        const enemyUnder = e.target.closest('.enemy-slot, #enemyGodHpDisplay, #enemyGodSide, .god-hp-display, .god-card-wrapper');
        if (enemyUnder) { const rect = enemyUnder.getBoundingClientRect(); attackIndicator.style.display = 'block'; attackIndicator.style.left = rect.left + rect.width / 2 - 18 + 'px'; attackIndicator.style.top = rect.top - 8 + 'px'; }
        else { attackIndicator.style.display = 'none'; }
      }
    });

    document.addEventListener('mouseup', (e) => {
      if (!dragTracker.active) return;
      svgLayer.style.display = 'none'; attackIndicator.style.display = 'none';
      const targetSlot = e.target.closest('.slot, .player-slot, #enemyGodHpDisplay, #enemyGodSide, .god-hp-display, .god-card-wrapper');
      if (targetSlot) { resolveDragAction(targetSlot); selectedAttacker = null; document.querySelectorAll('.card.selected-for-attack').forEach(c => c.classList.remove('selected-for-attack')); }
      dragTracker.active = false;
    });

    function updateArrow(currentX, currentY) {
      arrowLine.setAttribute('x1', dragTracker.startX);
      arrowLine.setAttribute('y1', dragTracker.startY);
      arrowLine.setAttribute('x2', currentX);
      arrowLine.setAttribute('y2', currentY);
    }
  }

  function playCardFromHandToSlot(targetSlot, handSelection) {
    if (handSelection.card.type !== 'Unit') return;
    const availableProv = state.provPool - state.provUsed;
    const { handIdx, card } = handSelection;
    if (availableProv < card.cost) { document.getElementById('logMessage').textContent = '⚠️ Not enough provisions!'; return; }
    const tgtRow = targetSlot.dataset.row; const tgtIdx = parseInt(targetSlot.dataset.idx);
    let targetArr = (tgtRow === 'melee') ? state.playerMelee : state.playerRanged;
    if (targetArr[tgtIdx]) return;
    state.hand.splice(handIdx, 1);
    if (!card.passive && card.type !== 'Spell') assignPassive(card);
    targetArr[tgtIdx] = { ...card, hasAttacked: false, playOrder: playerPlayOrderCounter++ };
    state.provUsed += card.cost;
    document.getElementById('logMessage').textContent = `Played ${card.name}${card.passive ? ' [' + card.passive + ']' : ''}`;
    renderAll(); checkAffordableCards();
  }

  function getEffectiveAtk(card) {
    let atk = card.atk;
    if (card.passive === 'Berserk' && card.hp < card.maxHp / 2) atk += 3;
    return atk;
  }

  function getShieldValue(card) {
    if (card.passive === 'Shield 1') return 1;
    if (card.passive === 'Shield 2') return 2;
    return 0;
  }

  function hasTauntOnBoard(arr1, arr2) {
    return [...arr1, ...arr2].some(c => c && c.passive === 'Taunt' && c.hp > 0);
  }

  function findTauntTarget(arr1, arr2) {
    for (let i = 0; i < arr1.length; i++) { if (arr1[i] && arr1[i].passive === 'Taunt' && arr1[i].hp > 0) return { arr: arr1, idx: i }; }
    for (let i = 0; i < arr2.length; i++) { if (arr2[i] && arr2[i].passive === 'Taunt' && arr2[i].hp > 0) return { arr: arr2, idx: i }; }
    return null;
  }

  function resolveDragAction(target) {
    if (!target) return;
    const availableProv = state.provPool - state.provUsed;
    spawnActionSpark(target.getBoundingClientRect().left + target.getBoundingClientRect().width/2, target.getBoundingClientRect().top, '⚡');

    if (dragTracker.type === 'combat' || (selectedAttacker && !selectedAttacker.isHandCard && target)) {
      let src = dragTracker.type === 'combat' ? dragTracker.origin : selectedAttacker;
      if (!src || !src.row || src.idx === undefined) return;
      const attackerArr = (src.row === 'melee') ? state.playerMelee : state.playerRanged;
      const attacker = attackerArr[src.idx];
      if (!attacker || attacker.hasAttacked) return;

            if (target.id === 'enemyGodHpDisplay' || target.closest('#enemyGodHpDisplay') || target.closest('#enemyGodSide')) {
        if (hasEnemyUnits()) { document.getElementById('logMessage').textContent = '⚠️ Clear all enemy units first!'; return; }
        const effAtk = getEffectiveAtk(attacker);
        state.enemyGodHp -= effAtk; attacker.hasAttacked = true; executeShake();
        showFloatingDamage(target, effAtk, '#ff6b35');
        document.getElementById('logMessage').textContent = `⚔️ ${attacker.name} hit enemy God for ${effAtk} damage!`;
        renderAll(); return;
      }
      
      if (target.classList.contains('enemy-slot')) {
        const tgtRow = target.dataset.row; const tgtIdx = parseInt(target.dataset.idx);
        const enemyArr = (tgtRow === 'melee') ? state.enemyMelee : state.enemyRanged;
        let enemyCard = enemyArr[tgtIdx];

        if (!enemyCard && hasEnemyUnits()) { document.getElementById('logMessage').textContent = '⚠️ Clear all enemy units first!'; return; }
        if (!enemyCard && !hasEnemyUnits()) {
          const effAtk = getEffectiveAtk(attacker);
          state.enemyGodHp -= effAtk; attacker.hasAttacked = true; executeShake();
          showFloatingDamage(target, effAtk, '#ff6b35');
          document.getElementById('logMessage').textContent = `⚔️ ${attacker.name} hit enemy God for ${effAtk} damage!`;
          renderAll(); return;
        }

        // TAUNT CHECK
        if (hasTauntOnBoard(state.enemyMelee, state.enemyRanged)) {
          let enemyHasTaunt = enemyCard && enemyCard.passive === 'Taunt' && enemyCard.hp > 0;
          if (!enemyHasTaunt) {
            document.getElementById('logMessage').textContent = '⚠️ An enemy has Taunt! You must attack it first.';
            return;
          }
        }

        const targetDom = target.querySelector('.card');
        const attackerSlotEl = document.querySelector(`.player-slot[data-row="${src.row}"][data-idx="${src.idx}"]`);
        const attackerDom = attackerSlotEl?.querySelector('.card');
        if (attackerDom) animateCardSlash(attackerDom);

        let effAtk = getEffectiveAtk(attacker);
        let shieldVal = getShieldValue(enemyCard);
        if (attacker.passive === 'Pierce') shieldVal = 0;

        if (shieldVal > 0) {
          shieldVal--;
          enemyCard.passive = shieldVal > 0 ? `Shield ${shieldVal}` : null;
          document.getElementById('logMessage').textContent = `🛡️ ${enemyCard.name}'s shield absorbed the hit!`;
          attacker.hasAttacked = true;
          targetDom?.classList.add('defend-flash'); executeShake(); renderAll(); return;
        }

        let attacks = attacker.passive === 'Double Strike' ? 2 : 1;
        for (let a = 0; a < attacks; a++) {
          if (!enemyCard || enemyCard.hp <= 0) break;
          if (effAtk >= enemyCard.hp) {
            document.getElementById('logMessage').textContent = `💥 ${attacker.name} destroyed ${enemyCard.name}!`;

              attacker.hp = Math.min(attacker.maxHp, attacker.hp + healAmt);
              spawnActionSpark(target.getBoundingClientRect().left, target.getBoundingClientRect().top, '💚');
            }
            if (enemyCard.passive && enemyCard.passive.startsWith('Thorns')) {
              const thornDmg = parseInt(enemyCard.passive.split(' ')[1]) || 1;
              attacker.hp -= thornDmg;
              showFloatingDamage(attackerSlotEl || target, thornDmg, '#ff8800');
              if (attacker.hp <= 0) { state.playerGrave.push({...attacker, hp: attacker.maxHp}); attackerArr[src.idx] = null; }
            }
            state.enemyGrave.push({...enemyCard, hp: enemyCard.maxHp});
            enemyArr[tgtIdx] = null;
            enemyCard = null;
            targetDom?.classList.add('burst-vanish'); executeShake();
          } else {
            enemyCard.hp -= effAtk; showFloatingDamage(target, effAtk);
            if (enemyCard.passive && enemyCard.passive.startsWith('Thorns')) {
              const thornDmg = parseInt(enemyCard.passive.split(' ')[1]) || 1;
              attacker.hp -= thornDmg;
              showFloatingDamage(attackerSlotEl || target, thornDmg, '#ff8800');
              if (attacker.hp <= 0) { state.playerGrave.push({...attacker, hp: attacker.maxHp}); attackerArr[src.idx] = null; break; }
            }
            targetDom?.classList.add('defend-flash'); executeShake();
          }

      }
    }

    if (dragTracker.type === 'play-unit') {
      const { handIdx, card } = dragTracker.origin;
      if (target.classList.contains('player-slot')) {
        if (availableProv < card.cost) { document.getElementById('logMessage').textContent = '⚠️ Not enough provisions!'; return; }

        state.hand.splice(handIdx, 1);
        if (!card.passive && card.type !== 'Spell') assignPassive(card);

        if (enemyArr[idx]) { state.provUsed += card.cost; state.hand.splice(handIdx, 1); enemyArr[idx].hp -= 3; showFloatingDamage(target, 3); if (enemyArr[idx].hp <= 0) { state.enemyGrave.push({...enemyArr[idx], hp: enemyArr[idx].maxHp}); enemyArr[idx] = null; target.querySelector('.card')?.classList.add('burst-vanish'); } executeShake(); document.getElementById('logMessage').textContent = '🔥 Spell dealt 3 damage!'; setTimeout(() => { renderAll(); checkAffordableCards(); }, 400); }

        if (playerArr[idx]) { state.provUsed += card.cost; state.hand.splice(handIdx, 1); playerArr[idx].atk += 2; playerArr[idx].hp += 2; playerArr[idx].maxHp += 2; document.getElementById('logMessage').textContent = `✨ Buffed ${playerArr[idx].name}!`; renderAll(); checkAffordableCards(); }
      } else if (card.effect === 'damage_6' && target.classList.contains('enemy-slot')) {
        const enemyArr = (target.dataset.row === 'melee') ? state.enemyMelee : state.enemyRanged; const idx = parseInt(target.dataset.idx);
        if (enemyArr[idx]) { state.provUsed += card.cost; state.hand.splice(handIdx, 1); enemyArr[idx].hp -= 6; showFloatingDamage(target, 6); if (enemyArr[idx].hp <= 0) { state.enemyGrave.push({...enemyArr[idx], hp: enemyArr[idx].maxHp}); enemyArr[idx] = null; target.querySelector('.card')?.classList.add('burst-vanish'); } executeShake(); document.getElementById('logMessage').textContent = '🔥🔥 Heavenly Strike dealt 6 damage!'; setTimeout(() => { renderAll(); checkAffordableCards(); }, 400); }

        state.provUsed += card.cost; state.hand.splice(handIdx, 1); [state.enemyMelee, state.enemyRanged].forEach(arr => { arr.forEach((c, i) => { if (c) { c.hp -= 2; if (c.hp <= 0) { state.enemyGrave.push({...c, hp: c.maxHp}); arr[i] = null; } } }); }); executeHeavyShake(); document.getElementById('logMessage').textContent = '🌋 Cataclysm dealt 2 damage to all enemies!'; setTimeout(() => { renderAll(); checkAffordableCards(); }, 400);
      } else if (card.effect === 'destroy_strongest' && target.classList.contains('enemy-slot')) {
        state.provUsed += card.cost; state.hand.splice(handIdx, 1); let strongest = null; let strongestArr = null; let strongestIdx = -1;
        [{arr: state.enemyMelee}, {arr: state.enemyRanged}].forEach(({arr}) => { arr.forEach((c, i) => { if (c && (!strongest || c.atk > strongest.atk)) { strongest = c; strongestArr = arr; strongestIdx = i; } }); });
        if (strongest) { state.enemyGrave.push({...strongest, hp: strongest.maxHp}); strongestArr[strongestIdx] = null; executeHeavyShake(); document.getElementById('logMessage').textContent = `💀 Dark Ritual destroyed ${strongest.name}!`; }
        setTimeout(() => { renderAll(); checkAffordableCards(); }, 400);
      } else if (card.effect === 'resurrect_2' && target.classList.contains('player-slot')) {

        setTimeout(() => { renderAll(); checkAffordableCards(); }, 400);
      }
    }
  }

  function getPassiveCssClass(card) {
    if (!card || !card.passive || !PASSIVES[card.passive]) return '';
    return PASSIVES[card.passive].cssClass || '';
  }

  function renderCardHTML(c, availableProv) {
    if (!c) return '';
    let isUnaffordable = availableProv !== undefined && c.cost > availableProv;
    let borderClass = c.type === 'Spell' ? 'spell-red-border' : getBorderClass(c.cost);
    let passiveClass = getPassiveCssClass(c);
    let passiveTypeClass = c.passive && PASSIVES[c.passive] ? PASSIVES[c.passive].type : '';
    let classes = `card ${c.type === 'Spell' ? 'spell-card' : ''} ${borderClass} ${c.hasAttacked ? 'has-attacked' : ''} ${isUnaffordable ? 'unaffordable' : ''} ${passiveClass}`;
    let passiveInfo = c.passive && PASSIVES[c.passive] ? PASSIVES[c.passive] : null;
    if (c.type === 'Spell') {
      return `<div class="${classes}"><div class="card-top-row"><span class="card-spell-icon">🔥</span><span class="card-prov-badge">🍗 ${c.cost}</span></div><div class="card-spell-desc-top">${c.desc}</div><div class="card-name-center">${c.name}</div></div>`;
    }
    return `<div class="${classes}"><div class="card-top-row"><span class="card-atk-badge">${c.atk !== undefined ? '⚔️ ' + c.atk : ''}</span><span class="card-hp-badge">${c.hp !== undefined && !isNaN(c.hp) ? '❤️ ' + c.hp : '❤️ ?'}</span><span class="card-prov-badge">🍗 ${c.cost}</span></div><div class="card-name-center">${c.name}</div>${passiveInfo ? `<div class="card-passive-badge ${passiveTypeClass}">${passiveInfo.icon} ${passiveInfo.name}</div>` : ''}</div>`;
  }

  function executeEnemyAITurn() {
    let enemyTime = 30;
    const enemyTimerEl = document.getElementById('enemyTimer');
    enemyTimerEl.textContent = enemyTime + 's';
    enemyTimerEl.classList.remove('critical-danger');
    if (state.enemyTimerInterval) clearInterval(state.enemyTimerInterval);
    state.enemyTimerInterval = setInterval(() => {
      enemyTime--;
      enemyTimerEl.textContent = enemyTime + 's';
      if (enemyTime <= 10) enemyTimerEl.classList.add('critical-danger');
      else enemyTimerEl.classList.remove('critical-danger');
      if (enemyTime <= 0) clearInterval(state.enemyTimerInterval);
    }, 1000);

    const allEnemyAttackers = [];
    state.enemyMelee.forEach((u, idx) => { if (u && !u.hasAttacked) allEnemyAttackers.push({ card: u, arr: state.enemyMelee, idx, row: 'melee' }); });
    state.enemyRanged.forEach((u, idx) => { if (u && !u.hasAttacked) allEnemyAttackers.push({ card: u, arr: state.enemyRanged, idx, row: 'ranged' }); });

    allEnemyAttackers.forEach(attackerInfo => {
      let u = attackerInfo.card;
      let tauntTarget = findTauntTarget(state.playerMelee, state.playerRanged);
      let targetArr, targetIdx;

      if (tauntTarget) {
        targetArr = tauntTarget.arr;
        targetIdx = tauntTarget.idx;
      } else {
        let allTargets = [];
        state.playerMelee.forEach((c, i) => { if (c) allTargets.push({ card: c, arr: state.playerMelee, idx: i }); });
        state.playerRanged.forEach((c, i) => { if (c) allTargets.push({ card: c, arr: state.playerRanged, idx: i }); });
        allTargets.sort((a, b) => a.card.hp - b.card.hp);
        if (allTargets.length > 0) {
          const best = allTargets[0];
          targetArr = best.arr;
          targetIdx = best.idx;
        }
      }

      if (targetArr && targetIdx !== undefined && targetArr[targetIdx]) {
        let target = targetArr[targetIdx];
        let effAtk = u.atk;
        if (u.passive === 'Berserk' && u.hp < u.maxHp / 2) effAtk += 3;
        let shieldVal = getShieldValue(target);
        if (u.passive === 'Pierce') shieldVal = 0;

        if (shieldVal > 0) {
          shieldVal--;
          target.passive = shieldVal > 0 ? `Shield ${shieldVal}` : null;
        } else {
          let attacks = u.passive === 'Double Strike' ? 2 : 1;
          for (let a = 0; a < attacks; a++) {
            if (!targetArr[targetIdx] || targetArr[targetIdx].hp <= 0) break;
            target = targetArr[targetIdx];
            if (effAtk >= target.hp) {
              if (target.passive && target.passive.startsWith('Thorns')) {
                const thornDmg = parseInt(target.passive.split(' ')[1]) || 1;
                u.hp -= thornDmg;
                if (u.hp <= 0) { state.enemyGrave.push({...u, hp: u.maxHp}); attackerInfo.arr[attackerInfo.idx] = null; break; }
              }
              if (u.passive && u.passive.startsWith('Lifesteal')) {
                const healAmt = parseInt(u.passive.split(' ')[1]) || 1;
                u.hp = Math.min(u.maxHp, u.hp + healAmt);
              }
              state.playerGrave.push({...target, hp: target.maxHp});
              targetArr[targetIdx] = null;
            } else {
              target.hp -= effAtk;
              if (target.hp <= 0) {
                if (target.passive && target.passive.startsWith('Thorns')) {
                  const thornDmg = parseInt(target.passive.split(' ')[1]) || 1;
                  u.hp -= thornDmg;
                  if (u.hp <= 0) { state.enemyGrave.push({...u, hp: u.maxHp}); attackerInfo.arr[attackerInfo.idx] = null; break; }
                }
                state.playerGrave.push({...target, hp: target.maxHp});
                targetArr[targetIdx] = null;
              }
            }
          }
        }
      } else {
        state.godHp -= u.atk;
        executeShake();
      }
      if (attackerInfo.arr[attackerInfo.idx]) attackerInfo.arr[attackerInfo.idx].hasAttacked = true;
    });

    if (state.godHp <= 0) { state.gameOver = true; renderAll(); triggerAutoOverlay('💀 Defeat'); clearInterval(state.enemyTimerInterval); enemyTimerEl.textContent = '30s'; return; }

    let enemyProvPool = state.provPerTurn + 2;
    let enemyProvUsed = 0;
    let cardsPlayed = 0;
    const maxCards = Math.min(3, enemyProvPool);

    while (cardsPlayed < maxCards && state.enemyDeckSize > 0) {
      const bronzePool = state.enemyFaction === 'heroes' ? BRONZE_UNITS_HEROES : BRONZE_UNITS_MONSTERS;
      const goldPool = state.enemyFaction === 'heroes' ? GOLD_UNITS_HEROES : GOLD_UNITS_MONSTERS;
      let card;
      const r = Math.random();
      if (r < 0.55) { card = { ...bronzePool[Math.floor(Math.random() * bronzePool.length)] }; }
      else if (r < 0.85) { card = { ...goldPool[Math.floor(Math.random() * goldPool.length)] }; }
      else { card = { ...SPELLS[Math.floor(Math.random() * SPELLS.length)] }; }

      if (card.cost + enemyProvUsed > enemyProvPool) break;

      assignPassive(card);
      if (card.type === 'Unit') {
        let arr = Math.random() < 0.5 ? state.enemyMelee : state.enemyRanged;
        let slot = arr.findIndex(s => s === null);
        if (slot === -1) { arr = arr === state.enemyMelee ? state.enemyRanged : state.enemyMelee; slot = arr.findIndex(s => s === null); }
        if (slot !== -1) {
          arr[slot] = { ...card, hasAttacked: false, playOrder: enemyPlayOrderCounter++ };
          enemyProvUsed += card.cost;
          state.enemyDeckSize--;
          cardsPlayed++;
        } else break;
      } else {
        if (card.effect === 'damage_3' || card.effect === 'damage_6') {
          let targetArr = state.playerMelee;
          let tIdx = targetArr.findIndex(c => c !== null);
          if (tIdx === -1) { targetArr = state.playerRanged; tIdx = targetArr.findIndex(c => c !== null); }
          if (tIdx !== -1) {
            const dmg = card.effect === 'damage_6' ? 6 : 3;
            targetArr[tIdx].hp -= dmg;
            if (targetArr[tIdx].hp <= 0) { state.playerGrave.push({...targetArr[tIdx], hp: targetArr[tIdx].maxHp}); targetArr[tIdx] = null; }
            enemyProvUsed += card.cost;
            state.enemyDeckSize--;
            cardsPlayed++;
          }
        } else if (card.effect === 'divine_buff') {
          let buffArr = state.enemyMelee;
          let bIdx = buffArr.findIndex(c => c !== null);
          if (bIdx === -1) { buffArr = state.enemyRanged; bIdx = buffArr.findIndex(c => c !== null); }
          if (bIdx !== -1) {
            buffArr[bIdx].atk += 2; buffArr[bIdx].hp += 2; buffArr[bIdx].maxHp += 2;
            enemyProvUsed += card.cost;
            state.enemyDeckSize--;
            cardsPlayed++;
          }
        } else { break; }
      }
    }

    if (state.deck.length && state.hand.length < 7) drawCard();
    state.playerMelee.forEach(u => { if(u) u.hasAttacked = false; });
    state.playerRanged.forEach(u => { if(u) u.hasAttacked = false; });

    state.playerMelee.forEach(u => { if (u && u.passive === 'Regen 1') { u.hp = Math.min(u.maxHp, u.hp + 1); } });
    state.playerRanged.forEach(u => { if (u && u.passive === 'Regen 1') { u.hp = Math.min(u.maxHp, u.hp + 1); } });
    state.enemyMelee.forEach(u => { if (u && u.passive === 'Regen 1') { u.hp = Math.min(u.maxHp, u.hp + 1); } });
    state.enemyRanged.forEach(u => { if (u && u.passive === 'Regen 1') { u.hp = Math.min(u.maxHp, u.hp + 1); } });

    state.round++;
    state.provPerTurn = Math.min(state.provPerTurn + 2, 10);
    state.provPool = Math.min(state.provPool + 2, 20);
    state.provUsed = 0;

    clearInterval(state.enemyTimerInterval);
    enemyTimerEl.textContent = '30s';
    enemyTimerEl.classList.remove('critical-danger');

    // Check if dice should trigger (before incrementing round)
    const diceRound = state.round + 1;
    
    state.round++;
    state.provPerTurn = Math.min(state.provPerTurn + 2, 10);
    state.provPool = Math.min(state.provPool + 2, 20);
    state.provUsed = 0;

    clearInterval(state.enemyTimerInterval);
    enemyTimerEl.textContent = '30s';
    enemyTimerEl.classList.remove('critical-danger');

    if (diceRound % 2 === 0 && diceRound >= 2) {
      setTimeout(() => {
        triggerDiceRoll();
        setTimeout(() => {
          renderAll(); startTimer(); triggerAutoOverlay('Your Turn'); checkAffordableCards();
        }, 6000);
      }, 500);
    } else {
      setTimeout(() => { renderAll(); startTimer(); triggerAutoOverlay('Your Turn'); checkAffordableCards(); }, 1000);
    }
  }
